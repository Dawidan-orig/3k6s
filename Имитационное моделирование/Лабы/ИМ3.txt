	SIMULATE
	INITIAL X$BUF,0
	INITIAL X$MAX,15

	GENERATE ,,,1,10
MET1	SAVEVALUE BUF+,1
MET2	ADVANCE 7
	TEST L X$BUF,X$MAX,MET2
	TRANSFER ,MET1

	GENERATE ,,,1,5
MET3	TEST G X$BUF,0
	SEIZE CPU
	SAVEVALUE BUF-,1
	ADVANCE 15,8
	RELEASE CPU

	TRANSFER .95,,MET3
	SAVEVALUE BUF,0
	TRANSFER ,MET3

	GENERATE 300000
	TERMINATE 1
	START 1
	END
